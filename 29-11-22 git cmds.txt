git init --- initializes to git repo and creates a .git folder
git status -- shows the status of current changes in the git folder
git add --- adds the changes from wsp to staging area
git rm --cached filename -- reverts changes from staging area to wsp
git restore --staged filename -- modifiled files are reverted from sa to wsp
git commit -m "comment" -- to commit file from SA to LR (default master branch)

git config --global user.name username-- to configure username for git installation on machine
git config --global user.email useremail-- to configure user email
git config --list --- to check username and email
git config --global --edit -----to edit username/useremail

git remote -v ---- to check any remote origin
git remote show origin --- to check any remote origin
git remote add origin originurl -- configures RR url
git remote rm origin -- removes origin url

git push origin master -- LR to RR push the chnages which are committed

git log -- shows all the commites in LR
git log -1
git log -2
git log -3 --- shows latest 3 commits
git log --oneline --- summerize all the commits
git show commitID -- details of the commits will be shown

git clone url -- downloads rr with history to local machine
git pull origin master -- downloads latest change which is done in RR to LR
